Firestore SweetFirestore Syntactic Sugar

import sweet from "firestore-sweet"
const db = sweet(firebase.firestore) // sweetened

const comments = await db.get(
  "users",
  "bob",
  "comments",
  ["likes", ">", 10],
  ["likes", "desc"],
  10
)
const fs = firebase.firestore()

// equivalent original firestore sdk code
const docs = await fs
  .collection("users")
  .doc("bob")
  .collection("comments")
  .where("likes", ">", 10)
  .orderBy("likes", "desc")
  .limit(10)
  .get()

let comments = []
docs.forEach((ss) => {
  comments.push(ss.data())
})
